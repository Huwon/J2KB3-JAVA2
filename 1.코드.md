    package pjtTestl
    
    public class MainClass{

        public static void main(String[] args){

            int i = 10;
            System.out.println("i = " + i);

            i = 100;
            System.out.println("i = " + i);

            i = 300;
            System.out.println("i = " + i);

            i = 30;
            System.out.println("i = " + i);
        }
    }


### 1-2. 결과

    i = 10
    i = 100
    i = 300
    i = 30
